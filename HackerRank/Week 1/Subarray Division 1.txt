Subarray Division 1

#include <stdio.h>
int birthday(int s_count, int* s, int d, int m) {
    int c = 0;
    for(int i=0; i<=s_count-m; i++){
        int sm = 0;
        for(int j=0; j<m; j++){
            sm += s[i+j];
        }
        if(sm == d){
            c++;
        }
    }
    return c;
}
int main() {
    int n;
    scanf("%d", &n); 
    int s[n];        
    for (int i = 0; i < n; i++) {
        scanf("%d", &s[i]);
    }
    int d, m;               
    scanf("%d %d", &d, &m);
    int r = birthday(n, s, d, m);
    printf("%d\n", r); 
    return 0;
}
