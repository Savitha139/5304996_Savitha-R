Min Max Sum

#include <stdio.h>
#include <limits.h>  

void miniMaxSum(int arr_count, int* arr) {
    long t = 0;
    int mx = INT_MIN;
    int mn = INT_MAX;
    
    for (int i = 0; i < arr_count; i++) {
        t += arr[i];
        if (arr[i] < mn) {
            mn = arr[i];
        }
        if (arr[i] > mx) {
            mx = arr[i];
        }
    }
    long min = t - mx;
    long max = t - mn;
    printf("%ld %ld\n", min, max);
}

int main() {
    int a[5];
    for (int i = 0; i < 5; i++) {
        scanf("%d", &a[i]);
    }
    miniMaxSum(5, a);
    return 0;
}
