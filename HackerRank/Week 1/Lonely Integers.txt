Lonely Integers

#include <stdio.h>
int lonelyinteger(int a_count, int* a) {
    int b = 0;
    for (int i = 0; i < a_count; i++) {
        b ^= a[i];
    }
    return b;
}

int main() {
    int n;
    scanf("%d", &n);
    int a[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    int k = lonelyinteger(n, a);
    printf("%d\n", k);
    return 0;
}
