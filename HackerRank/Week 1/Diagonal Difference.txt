Diagonal Difference

#include <stdio.h>
#include <stdlib.h>
int diagonalDifference(int arr_rows, int arr_columns, int** arr) {
    int a = 0;
    int b = 0;
    for(int i=0; i< arr_rows; i++){
        a += arr[i][i];
        b += arr[i][arr_columns-i-1];
    }
    return abs(a-b);
}
int main() {
    int n;
    scanf("%d", &n); 
    int** a = malloc(n * sizeof(int*));
    for (int i = 0; i < n; i++) {
        a[i] = malloc(n * sizeof(int));
        for (int j = 0; j < n; j++) {
            scanf("%d", &a[i][j]);
        }
    }
    int result = diagonalDifference(n, n, a);
    printf("%d\n", result);
    for (int i = 0; i < n; i++) {
        free(a[i]);
    }
    free(a);
    return 0;
}
