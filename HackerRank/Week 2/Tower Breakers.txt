Tower Breakers

#include <stdio.h>
int towerBreakers(int n, int m) {
     if(m == 1){
        return 2;
     }
     if(n%2==0){
        return 2;
     }
     return 1;
}
int main() {
    int a; 
    scanf("%d", &a);
    while (a--) {
        int n, m;
        scanf("%d %d", &n, &m);
        printf("%d\n", towerBreakers(n, m));
    }
}
