Caesar Cipher
#include <stdio.h>

char* caesarCipher(char* s, int k) {
    int l = strlen(s);
    k = k % 26;
    for(int i=0; i<l; i++){
        if(isalpha(s[i])){
            char b = isupper(s[i]) ? 'A' :'a';
            s[i] =((s[i]-b+k)%26)+b;
        }
    }
    return s;
}
int main() {
    int n, k;
    scanf("%d", &n); 
    char s[1001]; 
    scanf("%s", s); 
    scanf("%d", &k); 
    printf("%s\n", caesarCipher(s, k));
    return 0;
}
