Sherlock and Array

#include <stdio.h>
char* balancedSums(int arr_count, int* arr) {
    long t = 0;
    long lhs = 0;
    for(int i=0; i<arr_count; i++){
        t += arr[i];
    }
    for(int i=0; i<arr_count; i++){
        long rhs = t-lhs-arr[i];
        if(lhs==rhs){
            return "YES";
        }
        lhs += arr[i];
    }
    return "NO";
}
int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        scanf("%d", &n);
        int *a = malloc(n * sizeof(int));
        for (int i = 0; i < n; i++) {
            scanf("%d", &a[i]);
        }
        printf("%s\n", balancedSums(n, a));
        free(a);
    }
    return 0;
}
