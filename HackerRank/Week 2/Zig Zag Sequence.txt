Zig Zag Sequence

#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

void swap(int *a, int *b){
    int t = *a;
    *a = *b;
    *b = t;
}
int comparison(const void *a, const void *b){
    return (*(int*)a - *(int*)b);
}
int main() {
    int t;
    scanf("%d", &t);
    while(t--){
        int n;
        scanf("%d", &n);
        int a[n];
        for(int i=0; i<n; i++){
            scanf("%d", &a[i]);
        }
        qsort(a, n, sizeof(int), comparison);
        int m =(n-1)/2;
        swap(&a[m], &a[n-1]);
        int st = m+1, ed = n-2;
        while(st < ed){
            swap(&a[st], &a[ed]);
            st++;
            ed--;
        }
        
        for(int i=0; i<n; i++){
            if(i>0){
                printf(" ");
            }
            printf("%d", a[i]);
        }
        printf("\n");
    }    
    return 0;
}
