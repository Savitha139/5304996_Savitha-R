Max Min

#include <stdio.h>
int comp(const void *a, const void *b){
    return (*(int*)a - *(int*)b);
}
int maxMin(int k, int arr_count, int* arr) {
    qsort(arr, arr_count, sizeof(int), comp);
    int m = INT_MAX;
    for(int i=0; i<= arr_count-k; i++){
        int n = arr[i+k-1]-arr[i];
        if(n<m){
            m=n;
        }
    }
    return m;
}
int main() {
    int n, k;
    scanf("%d", &n);
    scanf("%d", &k);
    int a[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    printf("%d\n", maxMin(k, n, a));
    return 0;
}
